CREATE TABLE MEMBER (
	MEMBER_ID VARCHAR2(1000) PRIMARY KEY,
	PASSWORD VARCHAR2(1000) NOT NULL,
	NAME VARCHAR2(1000) NOT NULL,
	GENDERTYPE VARCHAR2(3) CHECK (GENDERTYPE IN ('M', 'F')),
	BIRTH DATE,
	EMAIL VARCHAR2(1000)
);
INSERT INTO MEMBER VALUES ('admin', '1234', '관리자', 'F', TO_DATE('1990-01-01', 'YYYY-MM-DD'), 'admin@example.com');

CREATE TABLE BOARD (
	BOARD_ID NUMBER PRIMARY KEY,
	TITLE VARCHAR2(3000) NOT NULL,
	CONTENTS VARCHAR2(4000) NOT NULL,
	MEMBER_ID VARCHAR2(1000),
	HIT NUMBER,
	CREATED_TIME DATE,
	CONSTRAINT FK_MEMID FOREIGN KEY(MEMBER_ID) REFERENCES MEMBER(MEMBER_ID)
);
CREATE SEQUENCE SEQ_BOARD NOCACHE;
INSERT INTO BOARD VALUES (SEQ_BOARD.NEXTVAL, 'test', '테스트 중...', 'admin', 1, SYSDATE);

CREATE TABLE REPLY (
	REPLY_ID NUMBER PRIMARY KEY,
	BOARD_ID NUMBER,
	MEMBER_ID VARCHAR2(1000),
	CONTENT VARCHAR2(4000) NOT NULL,
	CREATED_TIME DATE,
	CONSTRAINT FK_RE_BOARDID FOREIGN KEY(BOARD_ID) REFERENCES BOARD(BOARD_ID),
	CONSTRAINT FK_RE_MEMID FOREIGN KEY(MEMBER_ID) REFERENCES MEMBER(MEMBER_ID)
);
CREATE SEQUENCE SEQ_REPLY NOCACHE;
INSERT INTO REPLY VALUES (SEQ_REPLY.NEXTVAL, 1, 'admin', '리플 테스트 중...', SYSDATE);
